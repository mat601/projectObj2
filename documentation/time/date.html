	make_by_days (n: INTEGER_32)
			-- Set the current date with the number of days `n' since Origin.
		local
			i, j: INTEGER_32
			y, m, d: INTEGER_32
		do
			i := 4 * (n - 59) - 1
			j := (4 * div ((mod (i, 146097)), 4)) + 3
			y := (100 * (div (i, 146097))) + (div (j, 1461))
			i := (5 * (div (((mod (j, 1461)) + 4), 4))) - 3
			m := div (i, 153)
			d := div (((mod (i, 153)) + 5), 5)
			if m < 10 then
				m := m + 3
			else
				m := m - 9
				y := y + 1
			end
			y := y + 1600
			make (y, m, d)
		ensure
			days_set: days = n
		end
